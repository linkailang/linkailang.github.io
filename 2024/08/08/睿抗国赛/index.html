睿抗国赛

发表于2024-08-08|更新于2024-08-21
|字数总计:5.4k|阅读时长:27分钟|阅读量:

注意事项

比省赛要改

[!IMPORTANT]

参加实物比赛，多个充电区域，停留俩秒，放置物块，根据物块定位二维码进行放置（省赛没有），然后国赛多障碍物，需要自主避障，

当前任务

  • 先搞好仿真环境

  • 运行省赛cpp看看

  • 获取机械臂到ar码的坐标

  • 设定小车在地图上的坐标

省赛任务

1 机器人导航到工作台前方；

2 调用二次定位服务控制机器人对准工作台的AR码；

3 调用相对移动服务控制机器人走近工作台；

4 调用手臂控制服务控制手臂到达工作台上方观察位置；

5 调用AR码抓取服务，抓取工作台上的AR码；

6 调用AR码放置服务，将AR码模块放置到机器人自身buff上；

7 如果有多个AR码块，重复5-6步骤；

8 机器人导航到起始位置；

bool isTaskTimeout()
{
    static ros::Time start_time = ros::Time::now(); // 记录任务开始时间

    // 获取当前时间
    ros::Time current_time = ros::Time::now();

    // 计算任务已经运行的时间
    ros::Duration elapsed_time = current_time - start_time;

    // 如果任务已经运行超过10分钟，返回true，表示任务超时
    if (elapsed_time.toSec() > 600) // 10分钟 = 600秒
    {
        return true;
    }
    else
    {
        return false;
    }
}

bool WorkStationPickCallback1(int value)
{
    int taskStep = value;
    int taskState;
    std::vector<int> arIds;
    ros::Rate loop(1);

    while (ros::ok())
    {
        if (rei_cruise::RobotCruise::getInstance().GetTaskState() == rei_cruise::PAUSE)
        {
            rei_cruise::RobotCruise::getInstance().SetCallBackFuncStep(taskStep);
            break;
        }
        else if (rei_cruise::RobotCruise::getInstance().GetTaskState() == rei_cruise::STOP)
        {
            break;
        }

        if (isTaskTimeout())
        {
            ROS_WARN("Task timed out.");
            rei_cruise::RobotCruise::getInstance().Stop();
            break;
        }

        taskState = rei_cruise::RobotCruise::getInstance().GetTaskState();

        if (taskState == rei_cruise::TaskState::ACTIVE)
        {
            switch (taskStep)
            {
            case 0:
                if (oryxTask.ArAdjust(1))
                    taskStep = 1;
                else
                    rei_cruise::RobotCruise::getInstance().Pause();
                break;
            case 1:
                if (oryxTask.RelativeMove(0.18, 0, 0, "odom"))
                {
                    taskStep = 2;
                }
                else
                    rei_cruise::RobotCruise::getInstance().Pause();
                break;
            case 2:
                // 移动机械臂到工作站上方
                if (oryxTask.GotoPosition(workStationAbovePose_.linear.x, workStationAbovePose_.linear.y, workStationAbovePose_.linear.z))
                {
                    taskStep = 3;
                }
                else
                    rei_cruise::RobotCruise::getInstance().Pause();
                break;
            case 3:
                  // 获取 AR 标签 ID
                if (oryxTask.GetArIDs(arIds))
                {
                    taskStep = 4;
                }
                else
                {
                    rei_cruise::RobotCruise::getInstance().Pause();
                }
                break;
            case 4:
     // 检查是否检测到 AR 标签
    if (arIds.size() >= 1)
    {
        // 找到最小的 AR 标签 ID
        int minArId = arIds[0];
        for (size_t i = 1; i < arIds.size(); ++i)
        {
            if (arIds[i] < minArId)
            {
                minArId = arIds[i];
            }
        }
        
        if (oryxTask.PickPlace(minArId, workStationPoses_[0].linear.x, workStationPoses_[0].linear.y, workStationPoses_[0].linear.z))
        {
            
            taskStep = 5;
            return true; // 一次任务执行完毕
        }
        else
        {
            rei_cruise::RobotCruise::getInstance().Pause();
        }
    }
    else
    {
        rei_cruise::RobotCruise::getInstance().Pause();
    }
    break;
            case 5:
                return true;
                break;
            }
        }
        else if (taskState == rei_cruise::TaskState::STOP)
        {
            break;
        }
        loop.sleep();
    }
    return true;
}

// 获取当前小车位置的逻辑
geometry_msgs::Pose getCurrentPose() {
    tf::TransformListener listener;
    tf::StampedTransform transform;

    try {
        listener.waitForTransform("map", "base_link", ros::Time(0), ros::Duration(10.0));
        listener.lookupTransform("map", "base_link", ros::Time(0), transform);
    } catch (tf::TransformException &ex) {
        ROS_ERROR("%s", ex.what());
        ros::Duration(1.0).sleep();
    }

    geometry_msgs::Pose pose;
    pose.position.x = transform.getOrigin().x();
    pose.position.y = transform.getOrigin().y();
    pose.position.z = transform.getOrigin().z();
    pose.orientation.x = transform.getRotation().x();
    pose.orientation.y = transform.getRotation().y();
    pose.orientation.z = transform.getRotation().z();
    pose.orientation.w = transform.getRotation().w();
    return pose;
}

// 将小车移动到home点的逻辑
void moveRobotToHome() {
    ros::NodeHandle nh;
    ros::Publisher goal_pub = nh.advertise<geometry_msgs::PoseStamped>("/move_base_simple/goal", 10);
    geometry_msgs::PoseStamped goal;
    goal.header.frame_id = "map";
    goal.header.stamp = ros::Time::now();
    goal.pose = homePose; // 将homePose设为目标位置

    goal_pub.publish(goal);
    ROS_INFO("Moving to home position...");
}

// 调整到home点的回调函数
bool adjustToHome(int value) {
    geometry_msgs::Pose currentPose = getCurrentPose();
    ROS_INFO("Current Position -> x: %f, y: %f, theta: %f",
             currentPose.position.x, currentPose.position.y, tf::getYaw(currentPose.orientation));

    double dx = currentPose.position.x - homePose.position.x; // 与home位置的x偏差
    double dy = currentPose.position.y - homePose.position.y; // 与home位置的y偏差
    double dtheta = tf::getYaw(currentPose.orientation) - tf::getYaw(homePose.orientation); // 与home位置的角度偏差

    // 检查当前位置与home点的差距
    if (fabs(dx) > POSITION_TOLERANCE || fabs(dy) > POSITION_TOLERANCE || fabs(dtheta) > ORIENTATION_TOLERANCE) {
        moveRobotToHome();
    } else {
        ROS_INFO("Robot is correctly positioned at home.");
    }

    return true;
}

bool WorkStationRetrieveCallback1(int value)
{
    int taskStep = value;
    int taskState;
    std::vector<int> arIds;
    ros::Rate loop(1);
    // 指定要识别的目标 ID 号
    int targetArId = 8; 
    while (ros::ok())
    {
        if (rei_cruise::RobotCruise::getInstance().GetTaskState() == rei_cruise::PAUSE)
        {
            rei_cruise::RobotCruise::getInstance().SetCallBackFuncStep(taskStep);
            break;
        }
        else if (rei_cruise::RobotCruise::getInstance().GetTaskState() == rei_cruise::STOP)
        {
            break;
        }
        taskState = rei_cruise::RobotCruise::getInstance().GetTaskState();

        if (taskState == rei_cruise::TaskState::ACTIVE)
        {
            switch (taskStep)
            {
            case 0:
                if (oryxTask.ArAdjust(1,0.35))
                    taskStep = 1;
                else
                    rei_cruise::RobotCruise::getInstance().Pause();
                break;
            case 1:
                if (oryxTask.RelativeMove(0.10, 0, 0, "odom"))
                {
                    taskStep = 2;
                }
                else
                    rei_cruise::RobotCruise::getInstance().Pause();
                break;
            case 2:
                if (oryxTask.GotoPosition(retrieveStationAbovePose_.linear.x, retrieveStationAbovePose_.linear.y, retrieveStationAbovePose_.linear.z))
                {
                    taskStep = 3;
                }
                else
                    rei_cruise::RobotCruise::getInstance().Pause();
                break;
            case 3:
                if (oryxTask.GetArIDs(arIds))
                {
                    taskStep = 4;
                }
                else
                {
                    rei_cruise::RobotCruise::getInstance().Pause();
                }
                break;
            case 4:

                 if (std::find(arIds.begin(), arIds.end(), targetArId) != arIds.end())
                 {
                    if (oryxTask.PickPlace(targetArId, boxPose_[0].linear.x, boxPose_[0].linear.y, boxPose_[0].linear.z))
                     {
                        taskStep = 5;
                        return true; // 一次任务执行完毕
                      }
                    else
                    {
                        rei_cruise::RobotCruise::getInstance().Pause();
                    }
                  }
                  else
                  {
                     rei_cruise::RobotCruise::getInstance().Pause();
                  }
                break;
            case 5:
                return true;
                break;
            }
        }
        else if (taskState == rei_cruise::TaskState::STOP)
        {
            break;
        }
        loop.sleep();
    }
    return true;
}

int main(int argc, char **argv)
{
    // 初始化ROS节点
    ros::init(argc, argv, "cruise_test_node");
    ros::NodeHandle nh;
    // 初始化ORYXTask对象
    if (!oryxTask.Init(nh))
    {
        return 0;
    }
    // 初始化速度发布者
    velocity_pub = nh.advertise<geometry_msgs::Twist>("cmd_vel", 10);
    // 设置小车初始速度 (这里设定为线速度0.5 m/s，角速度0.0 rad/s)
    setInitialSpeed(0.5, 0.0, 0.0);
   // 在程序启动时初始化 TransformListener
    initTransformListener();
    // 设置箱子上方的位置
    boxAbovePose_.linear.x = 90.0;
    boxAbovePose_.linear.y = 140.0;
    boxAbovePose_.linear.z = 120.0;
    boxAbovePose_.angular.z = 0.0;
    // 重置机械臂到指定位置
    if (!oryxTask.ResetArm(boxAbovePose_.linear.x, boxAbovePose_.linear.y, 60, 0))
    {
        return 0;
    }
    // 设置工作站上方的位置
    workStationAbovePose_.linear.x = 90.0;
    workStationAbovePose_.linear.y = 140.0;
    workStationAbovePose_.linear.z = 120.0;
    workStationAbovePose_.angular.z = 0.0;
    // 设置取货站上方的位置
    retrieveStationAbovePose_.linear.x = 100.0;
    retrieveStationAbovePose_.linear.y = -180.0;
    retrieveStationAbovePose_.linear.z = 120.0;
    retrieveStationAbovePose_.angular.z = 0.0;
    // 临时位置变量
    geometry_msgs::Twist tmpPose;
    // 添加第一个蓝色buff的位置
    tmpPose.linear.x = 105;
    tmpPose.linear.y = 118;
    tmpPose.linear.z = 20;
    tmpPose.angular.z = 0.0;
    boxPose_.push_back(tmpPose);
    // 添加第二个蓝色buff的位置
    tmpPose.linear.x = 107;
    tmpPose.linear.y = 183;
    tmpPose.linear.z = 23;
    tmpPose.angular.z = 0.0;
    boxPose_.push_back(tmpPose);
    // 获取RobotCruise单例对象
    rei_cruise::RobotCruise &cruise = rei_cruise::RobotCruise::getInstance();
    if (!cruise.Init(nh))
        return 0;
    // 添加开始回调函数
    cruise.AddStartCallBack(StartCallback);
    if (cruise.AddCallBack("workstation_pick1", WorkStationPickCallback1))
    {
        ROS_INFO("Set callback WorkStationPickCallback1 success");
    }

    if (cruise.AddCallBack("workstation_pick2", WorkStationPickCallback2))
    {
        ROS_INFO("Set callback WorkStationPickCallback2 success");
    }

    std::this_thread::sleep_for(std::chrono::seconds(3));
    if (cruise.AddCallBack("workstation_retrieve1", WorkStationRetrieveCallback1))
    {
        ROS_INFO("Set callback WorkStationRetrieveCallback1 success");
    }

    if (cruise.AddCallBack("workstation_retrieve2", WorkStationRetrieveCallback2))
    {
        ROS_INFO("Set callback WorkStationRetrieveCallback2 success");
    }
    if (cruise.AddCallBack("workstation_Charge1", WorkStationCharge1))
    {
        ROS_INFO("Set callback WorkStationCharge success");
    }
    if (cruise.AddCallBack("workstation_Charge", WorkStationCharge))
    {
        ROS_INFO("Set callback WorkStationCharge success");
    }
    std::this_thread::sleep_for(std::chrono::seconds(3));
    // 运行导航任务
    cruise.RunNavTask();
    return 0;
}
